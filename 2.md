**1. Адаптивная структура (Desktop vs Mobile)**

**Desktop (ширина \> 1024px):**

-   Боковое меню слева (sidebar)

-   Сетка статистики в 4 колонки

-   Основной контент с карточками

-   Фиксированная шапка

**Mobile (ширина \< 1024px):**

-   Нижняя навигация (bottom nav)

-   Сетка статистики в 1-2 колонки

-   Вертикальный скролл

-   Гамбургер-меню (опционально)

import { CategoryChart } from \'@/components/dashboard/CategoryChart\'

import {

Package,

Boxes,

AlertTriangle,

Tag,

TrendingUp,

Clock

} from \'lucide-react\'

import { Skeleton } from \'@/components/ui/skeleton\'

export default function DashboardPage() {

const { data: stats, isLoading } = useQuery({

queryKey: \[\'dashboard-stats\'\],

queryFn: async () =\> {

const { containers } = await supabase

.from(\'containers\')

.select(\'id\', { count: \'exact\' })

const { items } = await supabase

.from(\'items\')

.select(\'id, quantity\', { count: \'exact\' })

const totalItems = items?.reduce((acc, item) =\> acc + item.quantity, 0)
\|\| 0

const { lowStock } = await supabase

.from(\'items\')

.select(\'id\')

.lte(\'quantity\', 2)

return {

containers: containers?.length \|\| 0,

items: totalItems,

lowStock: lowStock?.length \|\| 0,

},

})

const { data } = await supabase

.limit(5)

return data

},

})

if (isLoading) {

return (

\<div className=\"space-y-6\"\>

\<div className=\"grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4
gap-4\"\>

{/\* Заголовок \*/}

Добро пожаловать! 👋

\</h1\>

Обзор вашего склада

\</p\>

\</div\>

{/\* Статистика \*/}

\<div className=\"grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4
gap-4\"\>

\<StatCard

title=\"Контейнеры\"

value={stats?.containers \|\| 0}

icon={Boxes}

gradient=\"from-blue-500 to-blue-600\"

/\>

\<StatCard

\<StatCard
